001    package grails.plugins.mongodb;
002    
003    import com.google.code.morphia.Datastore;
004    import com.google.code.morphia.DatastoreImpl;
005    import com.google.code.morphia.Morphia;
006    import com.mongodb.DB;
007    import com.mongodb.DBCollection;
008    import com.mongodb.Mongo;
009    import com.mongodb.ServerAddress;
010    import groovy.util.ConfigObject;
011    import org.apache.commons.logging.Log;
012    import org.apache.commons.logging.LogFactory;
013    import org.codehaus.groovy.grails.commons.GrailsApplication;
014    import org.springframework.context.ApplicationContext;
015    
016    import java.net.UnknownHostException;
017    import java.util.ArrayList;
018    import java.util.Arrays;
019    import java.util.List;
020    import java.util.Map;
021    
022    /**
023     * should be registered with spring. provides access to
024     * all relevant instances for working with mongodb
025     *
026     * @author: Juri Kuehn
027     */
028    public class MongoHolderBean {
029      private static final Log log = LogFactory.getLog(MongoDomainClass.class);
030    
031      private Morphia morphia;
032      private DatastoreImpl datastore;
033    
034      /**
035       * Parses config options, creates database connection and morphia & datastore instances
036       *
037       * @param application
038       * @throws UnknownHostException
039       */
040      public MongoHolderBean(GrailsApplication application) throws UnknownHostException {
041        Mongo mongo;
042    
043        Map flatConfig = application.getConfig().flatten();
044    
045        String database = getConfigVar(flatConfig, "mongodb.databaseName", null);
046        if (database == null) database = getConfigVar(flatConfig, "mongodb.database", "test");
047    
048        List<String> replicaSets = null;
049        try {
050          replicaSets = (List<String>)((ConfigObject)application.getConfig().get("mongodb")).get("replicaSet");
051        } catch (Exception ignore) {}
052    
053        if (replicaSets != null) { // user replica sets
054          log.info("Creating MongoDB connection with replica sets " + replicaSets + " and database " + database);
055          List<ServerAddress> addressList = new ArrayList<ServerAddress>();
056          for (String addr : replicaSets) {
057            addressList.add(new ServerAddress(addr));
058          }
059          mongo = new Mongo(addressList);
060        } else { // use host port
061          String host = getConfigVar(flatConfig, "mongodb.host", "localhost");
062          int port = parsePortFromConfig(getConfigVar(flatConfig, "mongodb.port", "27017"), 27017);
063          log.info("Creating MongoDB connection to host " + host + ":" + port + " and database " + database);
064          mongo = new Mongo(host, port);
065        }
066    
067        morphia = new Morphia();
068        String username = getConfigVar(flatConfig,"mongodb.username",null);
069        char[] password = getCharArray(getConfigVar(flatConfig, "mongodb.password", null));
070        datastore = (DatastoreImpl)morphia.createDatastore(mongo, database,username,password);
071    
072        // init ObjectFactory
073        morphia.getMapper().getOptions().objectFactory = new MongoDomainObjectFactory(application);
074      }
075    
076      private char[] getCharArray(String configVar) {
077          return configVar == null ? null : configVar.toCharArray();
078      }
079    
080      private String getConfigVar(Map config, String key, String defaultValue) {
081        if (!config.containsKey(key)) {
082          log.info("MongoDB configuration option missing: '" + key + "'. Using default value '" + defaultValue + "'.");
083          return defaultValue;
084        }
085        return config.get(key).toString();
086      }
087    
088      private int parsePortFromConfig(String configVal, int defaultValue) {
089        int port;
090        try { // in case port is not a valid number
091          port = Integer.parseInt(configVal);
092        } catch (Exception e) {
093          log.info("MongoDB port invalid: '" + configVal + "' is not a number. Using default value " + defaultValue);
094          return defaultValue;
095        }
096    
097        return port;
098      }
099    
100      public Morphia getMorphia() {
101        return morphia;
102      }
103    
104      public Datastore getDatastore() {
105        return datastore;
106      }
107    
108      public Mongo getMongo() {
109        return datastore.getMongo();
110      }
111    
112      public DB getDb() {
113        return datastore.getDB();
114      }
115    
116      /**
117       * returns the collection that stores instances of the given class
118       * @param clazz
119       * @return
120       */
121      public DBCollection getCollection(Class clazz) {
122        return datastore.getCollection(clazz);
123      }
124    
125      public DBCollection getCollection(Object obj) {
126        return datastore.getCollection(obj);
127      }
128    }
