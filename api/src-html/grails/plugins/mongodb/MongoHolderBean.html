001package grails.plugins.mongodb;
002
003import com.google.code.morphia.Datastore;
004import com.google.code.morphia.DatastoreImpl;
005import com.google.code.morphia.Morphia;
006import com.mongodb.*;
007import groovy.util.ConfigObject;
008import org.apache.commons.logging.Log;
009import org.apache.commons.logging.LogFactory;
010import org.codehaus.groovy.grails.commons.GrailsApplication;
011import org.springframework.context.ApplicationContext;
012
013import java.net.UnknownHostException;
014import java.util.ArrayList;
015import java.util.Arrays;
016import java.util.List;
017import java.util.Map;
018
019/**
020 * should be registered with spring. provides access to
021 * all relevant instances for working with mongodb
022 *
023 * @author: Juri Kuehn
024 */
025public class MongoHolderBean {
026  private static final Log log = LogFactory.getLog(MongoDomainClass.class);
027
028  private Morphia morphia;
029  private DatastoreImpl datastore;
030
031  /**
032   * Parses config options, creates database connection and morphia & datastore instances
033   *
034   * @param application
035   * @throws UnknownHostException
036   */
037  public MongoHolderBean(GrailsApplication application) throws UnknownHostException {
038    Mongo mongo;
039
040    Map flatConfig = application.getConfig().flatten();
041
042    String database = getConfigVar(flatConfig, "mongodb.databaseName", null);
043    if (database == null) database = getConfigVar(flatConfig, "mongodb.database", "test");
044
045    List<String> replicaSets = null;
046    try {
047      replicaSets = (List<String>)((ConfigObject)application.getConfig().get("mongodb")).get("replicaSet");
048    } catch (Exception ignore) {}
049
050    MongoOptions mongoOptions = null;
051    try {
052      mongoOptions = (MongoOptions)((ConfigObject)application.getConfig().get("mongodb")).get("options");
053    } catch (Exception ignore) { }
054    if (mongoOptions == null) mongoOptions = new MongoOptions();
055
056    if (replicaSets != null) { // user replica sets
057      log.info("Creating MongoDB connection with replica sets " + replicaSets + " and database " + database);
058      List<ServerAddress> addressList = new ArrayList<ServerAddress>();
059      for (String addr : replicaSets) {
060        addressList.add(new ServerAddress(addr));
061      }
062
063      mongo = new Mongo(addressList, mongoOptions);
064    } else { // use host port
065      String host = getConfigVar(flatConfig, "mongodb.host", "localhost");
066      int port = parsePortFromConfig(getConfigVar(flatConfig, "mongodb.port", "27017"), 27017);
067      log.info("Creating MongoDB connection to host " + host + ":" + port + " and database " + database);
068
069      mongo = new Mongo(new ServerAddress(host, port), mongoOptions);
070    }
071
072    morphia = new Morphia();
073    String username = getConfigVar(flatConfig,"mongodb.username",null);
074    char[] password = getCharArray(getConfigVar(flatConfig, "mongodb.password", null));
075    datastore = (DatastoreImpl)morphia.createDatastore(mongo, database,username,password);
076
077    // init ObjectFactory
078    morphia.getMapper().getOptions().objectFactory = new MongoDomainObjectFactory(application);
079  }
080
081  private char[] getCharArray(String configVar) {
082      return configVar == null ? null : configVar.toCharArray();
083  }
084
085  private String getConfigVar(Map config, String key, String defaultValue) {
086    if (!config.containsKey(key)) {
087      log.info("MongoDB configuration option missing: '" + key + "'. Using default value '" + defaultValue + "'.");
088      return defaultValue;
089    }
090    return config.get(key).toString();
091  }
092
093  private int parsePortFromConfig(String configVal, int defaultValue) {
094    int port;
095    try { // in case port is not a valid number
096      port = Integer.parseInt(configVal);
097    } catch (Exception e) {
098      log.info("MongoDB port invalid: '" + configVal + "' is not a number. Using default value " + defaultValue);
099      return defaultValue;
100    }
101
102    return port;
103  }
104
105  public Morphia getMorphia() {
106    return morphia;
107  }
108
109  public Datastore getDatastore() {
110    return datastore;
111  }
112
113  public Mongo getMongo() {
114    return datastore.getMongo();
115  }
116
117  public DB getDb() {
118    return datastore.getDB();
119  }
120
121  /**
122   * returns the collection that stores instances of the given class
123   * @param clazz
124   * @return
125   */
126  public DBCollection getCollection(Class clazz) {
127    return datastore.getCollection(clazz);
128  }
129
130  public DBCollection getCollection(Object obj) {
131    return datastore.getCollection(obj);
132  }
133}
