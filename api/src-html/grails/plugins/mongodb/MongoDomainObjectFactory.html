001package grails.plugins.mongodb;
002
003import com.google.code.morphia.mapping.DefaultCreator;
004import com.google.code.morphia.mapping.MappedField;
005import com.google.code.morphia.mapping.Mapper;
006import com.mongodb.DBObject;
007import org.apache.commons.logging.Log;
008import org.apache.commons.logging.LogFactory;
009import org.codehaus.groovy.grails.commons.GrailsApplication;
010import org.springframework.context.ApplicationContext;
011
012/**
013 * @author: Juri Kuehn
014 */
015public class MongoDomainObjectFactory extends DefaultCreator {
016
017  private static final Log log = LogFactory.getLog(MongoDomainClass.class);
018
019  private GrailsApplication application;
020
021  public MongoDomainObjectFactory(GrailsApplication application) {
022    this.application = application;
023  }
024
025  @Override
026  public Object createInstance(Class clazz) {
027    return getBean(clazz.getName());
028  }
029
030  @Override
031  public Object createInstance(Class clazz, DBObject dbObj) {
032    String className = getClassName(dbObj);
033                if (className != null) {
034      return getBean(className);
035    }
036
037    return createInstance(clazz);
038  }
039
040  @Override
041  public Object createInstance(Mapper mapr, MappedField mf, DBObject dbObj) {
042    String className = getClassName(dbObj);
043                if (className != null) {
044      return getBean(className);
045    }
046
047    Class c = mf.isSingleValue() ? mf.getConcreteType() : mf.getSubClass();
048                return createInstance(c);
049  }
050
051  private String getClassName(DBObject dbObj) {
052    // see if there is a className value
053    String className = (String) dbObj.get(Mapper.CLASS_NAME_FIELDNAME);
054    if (className != null) {
055      return className;
056    }
057    return null;
058  }
059
060  private Object getBean(String clazz) {
061    return application.getMainContext().getBean(clazz);
062  }
063
064
065}
