001package grails.plugins.mongodb;
002
003import com.google.code.morphia.annotations.Entity;
004import org.codehaus.groovy.grails.commons.ArtefactHandlerAdapter;
005import org.codehaus.groovy.grails.commons.GrailsClass;
006
007/**
008 * Author: Juri Kuehn
009 * Date: 30.05.2010
010 */
011public class MongoDomainClassArtefactHandler extends ArtefactHandlerAdapter {
012
013    public static String TYPE = "MongoDomain"; // dont change, plugin relies on this artefact name
014
015    public MongoDomainClassArtefactHandler() {
016      super(TYPE, MongoDomainClass.class, MongoDomainClass.class, null);
017    }
018
019    public boolean isArtefactClass(Class clazz) {
020        return isMongoDomainClass(clazz);
021    }
022
023    @SuppressWarnings("unchecked")
024    public static boolean isMongoDomainClass(Class clazz) {
025        return clazz != null && clazz.getAnnotation(Entity.class) != null;
026    }
027
028    public GrailsClass newArtefactClass(Class artefactClass) {
029        return new MongoDomainClass(artefactClass);
030    }
031}
